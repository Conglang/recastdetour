470  inline dtPolyRef encodePolyId(unsigned int salt, unsigned int it, unsigned int ip) const
471  {
472  return ((dtPolyRef)salt << (m_polyBits+m_tileBits)) | ((dtPolyRef)it << m_polyBits) | (dtPolyRef)ip;
473  }
474 
482  inline void decodePolyId(dtPolyRef ref, unsigned int& salt, unsigned int& it, unsigned int& ip) const
483  {
484  const dtPolyRef saltMask = ((dtPolyRef)1<<m_saltBits)-1;
485  const dtPolyRef tileMask = ((dtPolyRef)1<<m_tileBits)-1;
486  const dtPolyRef polyMask = ((dtPolyRef)1<<m_polyBits)-1;
487  salt = (unsigned int)((ref >> (m_polyBits+m_tileBits)) & saltMask);
488  it = (unsigned int)((ref >> m_polyBits) & tileMask);
489  ip = (unsigned int)(ref & polyMask);
490  }
491 
496  inline unsigned int decodePolyIdSalt(dtPolyRef ref) const
497  {
498  const dtPolyRef saltMask = ((dtPolyRef)1<<m_saltBits)-1;
499  return (unsigned int)((ref >> (m_polyBits+m_tileBits)) & saltMask);
500  }
501 
506  inline unsigned int decodePolyIdTile(dtPolyRef ref) const
507  {
508  const dtPolyRef tileMask = ((dtPolyRef)1<<m_tileBits)-1;

516  inline unsigned int decodePolyIdPoly(dtPolyRef ref) const
