91 template <typename T>
92 dtParametrizedBehavior<T>::dtParametrizedBehavior(unsigned preAllocationSize)
93  : m_size(preAllocationSize)
94 {
95  if (m_size == 0)
96  m_agentsParams = 0;
97  else
98  {
99  m_agentsParams = (Node*) dtAlloc(sizeof(Node) * m_size, DT_ALLOC_PERM);
100 
101  for (unsigned i = 0; i < m_size; ++i)
102  {
103  new(&m_agentsParams[i]) Node();
104  m_agentsParams[i].next = 0;
105  m_agentsParams[i].id = UINT_MAX;
106  }
107  }
108 }

110 template <typename T>
111 dtParametrizedBehavior<T>::~dtParametrizedBehavior()
112 {
113  for (unsigned i = 0; i < m_size; ++i)
114  {
115  Node* n = &m_agentsParams[i];
116 
117  while (n->next)
118  {
119  Node* toDelete = n->next;
120  n->next = n->next->next;
121  toDelete->~Node();
122  dtFree(toDelete);
123  }
124  }
125 
126  dtFree(m_agentsParams);
127  m_agentsParams = 0;
128 }

130 template <typename T>
131 T* dtParametrizedBehavior<T>::getBehaviorParams(unsigned id) const
132 {
133  if (m_size == 0)
134  return 0;
135 
136  unsigned index = id % m_size;
137 
138  if (index >= m_size)
139  return 0;
140 
141  Node* head = &m_agentsParams[index];
142 
143  if (head->id == id && head->id < UINT_MAX)
144  return &head->value;
145 
146  while (head->next)
147  {
148  head = head->next;
149 
150  if (head->id == id && head->id < UINT_MAX)
151  return &head->value;
152  }
153 
154  void* mem = dtAlloc(sizeof(Node), DT_ALLOC_PERM);
155 
156  if (mem == 0)
157  return 0;
158 
159  Node* newNode = new(mem) Node;
160 
161  newNode->next = 0;
162  newNode->id = id;
163  head->next = newNode;
164 
165  return &newNode->value;
166 }
