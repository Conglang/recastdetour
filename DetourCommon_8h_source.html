282 inline float dtVnormalize(float* v)
283 {
284  float d;
285  float length = dtVlen(v);
286 
287  if (length < EPSILON)
288  d = 0.f;
289  else
290  // Because of floating precision problems on some compilers, we have to redo the computation
291  d = 1.0f / dtSqrt(dtSqr(v[0]) + dtSqr(v[1]) + dtSqr(v[2]));
292 
293  v[0] *= d;
294  v[1] *= d;
295  v[2] *= d;
296 
297  return length;
298 }

469 inline unsigned int dtIlog2(unsigned int v)
470 {
471  unsigned int r;
472  unsigned int shift;
473  r = (v > 0xffff) << 4; v >>= r;
474  shift = (v > 0xff) << 3; v >>= shift; r |= shift;
475  shift = (v > 0xf) << 2; v >>= shift; r |= shift;
476  shift = (v > 0x3) << 1; v >>= shift; r |= shift;
477  r |= (v >> 1);
478  return r;
479 }
